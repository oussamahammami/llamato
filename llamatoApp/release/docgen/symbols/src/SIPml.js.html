228 /**
229 Initialize the engine. <b>You must call this function before any other.</b>.
230 @param {CallbackFunction} [readyCallback] Optional callback function to call when the stack finish initializing and become ready.
231 @param {CallbackFunction} [errorCallback] Optional callback function to call when initialization fails.
232 
233 @example
234 SIPml.init(function(e){ console.info('engine is ready'); }, function(e){ console.info('Error: ' + e.message); });
235 @static
236 */
237 SIPml.init = function (successCallback, errorCallback) {
238     if (!SIPml.b_initialized && !SIPml.b_initializing) {
239         SIPml.b_initializing = true;
240         tsk_utils_init_webrtc();
241 
242         tsk_utils_log_info('User-Agent=' + (navigator.userAgent || "unknown"));
243 
244         SIPml.b_have_media_stream = tsk_utils_have_stream();
245         SIPml.b_webrtc_supported = tsk_utils_have_webrtc();
246         SIPml.b_webrtc4all_supported = tsk_utils_have_webrtc4all();
247         SIPml.s_webrtc4all_version = tsk_utils_webrtc4all_get_version();
248         SIPml.s_navigator_friendly_name = tsk_utils_get_navigator_friendly_name();
249         SIPml.s_system_friendly_name = tsk_utils_get_system_friendly_name();
250 
251         // prints whether WebSocket is supported
252         tsk_utils_log_info("WebSocket supported = " + (SIPml.isWebSocketSupported() ? "yes" : "no"));
253 
254         // check webrtc4all version
255         if (tsk_utils_have_webrtc4all()) {
256             tsk_utils_log_info("WebRTC type = " + WebRtc4all_GetType() + " version = " + tsk_utils_webrtc4all_get_version());
257             if (SIPml.s_webrtc4all_version != '1.35.981') {
258                 SIPml.b_webrtc4all_plugin_outdated = true;
259             }
260         }
261 
262         // prints navigator friendly name
263         tsk_utils_log_info("Navigator friendly name = " + SIPml.s_navigator_friendly_name);
264 
265         // gets navigator version
266         if(SIPml.s_navigator_friendly_name == 'ie'){
267             var re = new RegExp("MSIE ([0-9]{1,}[\.0-9]{0,})");
268             if (re.exec(navigator.userAgent) != null) {
269                 SIPml.s_navigator_version = RegExp.$1;
270             }
271         }        
272 
273         // prints OS friendly name
274         tsk_utils_log_info("OS friendly name = " + SIPml.s_system_friendly_name);
275         // prints support for WebRTC (native or plugin)
276         tsk_utils_log_info("Have WebRTC = " + (tsk_utils_have_webrtc() ? "yes" : "false"));
277         // prints support for getUserMedia
278         tsk_utils_log_info("Have GUM = " + (tsk_utils_have_stream() ? "yes" : "false"));
279 
280         // checks for WebRTC support
281         if (!tsk_utils_have_webrtc()) {
282             // is it chrome?
283             if (SIPml.s_navigator_friendly_name == "chrome") {
284                 SIPml.b_navigator_outdated = true;
285                 return;
286             }
287 
288             // for now the plugins (WebRTC4all only works on Windows)
289             if (SIPml.s_system_friendly_name == 'win' || SIPml.s_system_friendly_name == 'windows') {
290                 // Internet explorer
291                 if (SIPml.s_navigator_friendly_name == 'ie') {
292                     // Check for IE version 
293                     var rv = -1;
294                     var ua = navigator.userAgent;
295                     var re = new RegExp("MSIE ([0-9]{1,}[\.0-9]{0,})");
296                     if (re.exec(ua) != null) {
297                         rv = parseFloat(RegExp.$1);
298                     }
299                     if (rv < 9.0) {
300                         SIPml.b_navigator_outdated = true;
301                         return;
302                     }
303 
304                     // break page loading ('window.location' won't stop JS execution)
305                     if (!tsk_utils_have_webrtc4all()) {
306                         return;
307                     }
308                 }
309             }
310         }
311 
312         if(SIPml.b_webrtc_supported && SIPml.b_have_media_stream){
313             SIPml.b_initialized = true;
314             SIPml.b_initializing = false;
315             tsk_utils_log_info("Engine initialized");
316             if(successCallback){
317                 successCallback({});
318             }
319         }
320         else{
321             if(errorCallback){
322                 var s_description = !SIPml.b_webrtc_supported ? "WebRTC not supported" : (!SIPml.b_have_media_stream ? "getUserMedia not supported" : "Internal error");
323                 errorCallback({description: s_description});
324             }
325         }
326     }
327 }

417     if (!o_listener) {
418         throw new Error("ERR_INVALID_PARAMETER_VALUE: 'listener' must not be null");
419     }
420     if (!o_type) {
421         throw new Error("ERR_INVALID_PARAMETER_VALUE: 'type' must not be null");
422     }
423     if(!(o_type instanceof String || typeof o_type == "string" || o_type instanceof Array)){
424         throw new Error("ERR_INVALID_PARAMETER_TYPE: 'type' must be a string or array");
425     }
426     
427     if(o_type instanceof Array){
428         var This = this;
429         o_type.forEach(function (s_type) {
430             if (!tsk_string_is_null_or_empty(s_type) && tsk_string_is_string(s_type)) {
431                 This.ao_listeners[s_type] = o_listener;
432             }
433         });
434     }
435     else{
436         this.ao_listeners[o_type] = o_listener;
437     }
438 }
